  def test_parse_ascii_8bit
    assert_equal @parsed, JSON.parse(@utf_8_ascii_8bit)
    assert_equal @parsed, JSON.parse(@utf_16be_ascii_8bit)
    assert_equal @parsed, JSON.parse(@utf_16le_ascii_8bit)

  end
