  def test_comments
    json = "{\n  \"key1\":\"value1\", // eol comment\n  \"key2\":\"value2\"  /* multi line\n                    *  comment */,\n  \"key3\":\"value3\"  /* multi line\n                    // nested eol comment\n                    *  comment */\n}\n"
    assert_equal(
      { "key1" => "value1", "key2" => "value2", "key3" => "value3" },
      parse(json))
    json = "{\n  \"key1\":\"value1\"  /* multi line\n                    // nested eol comment\n                    /* illegal nested multi line comment */\n                    *  comment */\n}\n"
    assert_raises(ParserError) { parse(json) }
    json = "{\n  \"key1\":\"value1\"  /* multi line\n                   // nested eol comment\n                   closed multi comment */\n                   and again, throw an Error */\n}\n"
    assert_raises(ParserError) { parse(json) }
    json = "{\n  \"key1\":\"value1\"  /*/*/\n}\n"
    assert_equal({ "key1" => "value1" }, parse(json))
  end
